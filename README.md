void setup() {
  Serial.begin(9600);
  WiFi.begin(WIFISSID, PASSWORD);
  mlx.begin();
  pinMode(14, INPUT); // Setup for leads off detection LO +
  pinMode(12, INPUT); // Setup for leads off detection LO -
  pinMode (buttonPin, INPUT);
  oled.begin(SSD1306_SWITCHCAPVCC, OLED_Address);
  oled.drawBitmap(0, 0, myBitmap, 128, 64, WHITE);
  oled.display();
  delay(1000);
  oled.setCursor(20,0);
  oled.setTextSize(1);
  oled.print("MENGHUBUNGKAN JARINGAN");
  oled.display();
  delay(1000);
  oled.clearDisplay();

  //cek status koneksi pada wifi
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }

  //cek komunikasi ubidots
  client.setServer(mqttBroker, 1883);
  client.setCallback(callback);
}


void loop() 
{
  button_suhu();
  ecg();
  oled_monitor();
  serial();

  //cek koneksi internet untuk upload data
  if (!client.connected()) {
    reconnect();
  }
  
  upload();

}
